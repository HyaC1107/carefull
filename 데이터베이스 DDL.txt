-- public.set_push_tokens_updated_at function

CREATE OR REPLACE FUNCTION public.set_push_tokens_updated_at()
 RETURNS trigger
 LANGUAGE plpgsql
AS $function$
BEGIN
  NEW.updated_at = CURRENT_TIMESTAMP;
  RETURN NEW;
END;
$function$;

-- Table Triggers

create trigger trg_push_tokens_updated_at before
update
    on
    public.push_tokens for each row execute function set_push_tokens_updated_at();
